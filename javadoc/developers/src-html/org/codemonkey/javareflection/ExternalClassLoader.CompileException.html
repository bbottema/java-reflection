001    package org.codemonkey.javareflection;
002    
003    import java.io.DataInputStream;
004    import java.io.File;
005    import java.io.FileInputStream;
006    import java.io.FileNotFoundException;
007    import java.io.IOException;
008    import java.util.HashMap;
009    import java.util.Map;
010    
011    import javax.script.ScriptException;
012    
013    /**
014     * A toolkit that can read and compile .java sourcefiles on the fly in runtime. This <code>class loader</code> caches loaded classes for
015     * repeated requests. Also, if a .class file already exists for a specific .java sourcefile, the classloader will compare dates and see if
016     * the .java needs recompiling. <br />
017     * <br />
018     * A basepath can be specified for the classloader to look in for sourcefiles.
019     */
020    public final class ExternalClassLoader extends ClassLoader {
021            /**
022             * base path used to locate [packaged] classes in
023             */
024            private String basepath;
025    
026            /**
027             * List of classes. These are non-instances of a class, of which instances can be spawn.
028             */
029            protected final Map<String, Class<?>> classes;
030    
031            /**
032             * The exception that was thrown when the class was actually found but some other error occurred.
033             */
034            protected CompileException exception;
035    
036            /**
037             * Constructor which initializes all properties.
038             */
039            public ExternalClassLoader() {
040                    super(ExternalClassLoader.class.getClassLoader());
041                    classes = new HashMap<String, Class<?>>();
042            }
043    
044            /**
045             * Loads a class from the classes cache if available. Delegates to {@link #findClass(String)} otherwise.
046             * 
047             * @see ClassLoader#loadClass(String)
048             */
049            @Override
050            public final Class<?> loadClass(final String className)
051                            throws ClassNotFoundException {
052                    final Class<?> c = classes.get(className);
053                    return (c != null) ? c : findClass(className);
054            }
055    
056            /**
057             * Loads a classfile from file in the following order:
058             * <ol>
059             * <li>looks for the java source and if available, checks whether it needs to be compiled</li>
060             * <li>if no .java or .class file found, try to find it in the VM</li>
061             * </ol>
062             * If the class ultimately wasn't found a <code>ClassNotFoundException</code> is being thrown. If class was found, but an error
063             * occurred, store exception for later review and return <code>null</code>. This is done so because the overridden method can't throw an
064             * exception other than <code>ClassNotFoundException</code>.
065             * 
066             * @param className The path and name to the classfile.
067             * @return The requested class reference.
068             * @throws ClassNotFoundException
069             */
070            @Override
071            public final Class<?> findClass(final String className)
072                            throws ClassNotFoundException {
073                    exception = null;
074                    try {
075                            // try to load, cache and return the class from compiled .class file
076                            checkForFile(basepath, className);
077    
078                            // if a classfile is available then load it
079                            final Class<?> c = classes.get(className);
080                            if (c != null) {
081                                    return c;
082                            } else {
083                                    throw new ClassNotFoundException();
084                            }
085                    } catch (final ClassNotFoundException e) {
086                            // - see if class can be found in the VM
087                            // - this check must come last, else you won't detect whether a classfile has been recompiled
088                            final Class<?> c = findSystemClass(className);
089                            classes.put(className, c);
090                            return c;
091                    } catch (final IOException e) {
092                            exception = new CompileException(e.getMessage(), e);
093                            return null;
094                    } catch (final CompileException e) {
095                            exception = e;
096                            return null;
097                    }
098            }
099    
100            /**
101             * Reads and compiles the sourcefile from filesystem and creates the class inside the VM.
102             * 
103             * @param className The path and name to the .java sourcefile.
104             * @throws CompileException
105             * @throws IOException
106             */
107            private void checkForFile(final String classPath, final String className)
108                            throws CompileException, IOException {
109                    // figure paths...
110                    final String resource = className.replace("..", "||").replace('.', File.separatorChar).replace("||", "..");
111                    final File javaSource = new File(classPath + File.separatorChar + resource + ".java");
112                    final File javaClass = new File(classPath + File.separatorChar + resource + ".class");
113    
114                    new File(basepath).getAbsolutePath();
115                    javaSource.getAbsolutePath();
116                    final String absoluteClassPath = javaClass.getAbsolutePath();
117    
118                    // see if there is a javafile and classfile
119                    if (javaSource.exists()) {
120                            // if classfile available, delete if outdated
121                            if (javaClass.exists()) {
122                                    // determine if the java sourcefile has been modified since last compile
123                                    // else check if the .class file has already been loaded
124                                    if (javaSource.lastModified() > javaClass.lastModified()) {
125                                            if (!javaClass.delete()) {
126                                                    throw new CompileException("runtime compiler: unable to removed outdated .class file");
127                                            }
128                                            classes.remove(className);
129                                    } else if (classes.get(className) == null) {
130                                            classes.put(className, loadClass(absoluteClassPath, className));
131                                    }
132                            }
133    
134                            // if no classfile available or became outdated
135                            if (!javaClass.exists()) {
136                                    final int status = 0;// Main.compile(args);
137    
138                                    // load compiled classfile and cache it or return error that occured during compiling
139                                    switch (status) {
140                                    case 0:
141                                            classes.put(className, loadClass(absoluteClassPath, className));
142                                            break;
143                                    case 1:
144                                            throw new CompileException("runtime compiler: ERROR");
145                                    case 2:
146                                            throw new CompileException("runtime compiler: CMDERR");
147                                    case 3:
148                                            throw new CompileException("runtime compiler: SYSERR");
149                                    case 4:
150                                            throw new CompileException("runtime compiler: ABNORMAL");
151                                    default:
152                                            throw new CompileException("Compile status: Unknown exit status");
153                                    }
154                            }
155                    }
156            }
157    
158            /**
159             * Reads the classfile from filesystem.
160             * 
161             * @param classPath The path and name to the classfile.
162             * @param className The name of the class to use as key for the Class reference value.
163             * @throws FileNotFoundException
164             * @throws IOException
165             */
166            private Class<?> loadClass(final String classPath, final String className)
167                            throws IOException {
168                    // get data from file
169                    final File f = new File(classPath);
170                    final int size = (int) f.length();
171                    final byte buff[] = new byte[size];
172                    final DataInputStream dis = new DataInputStream(new FileInputStream(f));
173                    dis.readFully(buff);
174                    dis.close();
175                    // convert data into class
176                    return defineClass(className, buff, 0, buff.length, null);
177            }
178    
179            /**
180             * Parameterized exception used when Java's runtime compiler fails to compile a Java source file.
181             * 
182             * @author Benny Bottema
183             */
184            public static class CompileException extends ScriptException {
185                    private static final long serialVersionUID = -7210219718456902667L;
186    
187                    /**
188                     * @param reason The description of the cause of the exception.
189                     */
190                    public CompileException(final String reason) {
191                            super(reason);
192                    }
193    
194                    /**
195                     * Used to create an exception with a copies stacktrace (using {@link #setStackTrace(StackTraceElement[])}).
196                     * 
197                     * @param reason The description of the cause of the exception.
198                     * @param cause A thrown exception that is the cause.
199                     */
200                    public CompileException(final String reason, final Throwable cause) {
201                            super(reason + "\n      " + cause.toString());
202                            setStackTrace(cause.getStackTrace());
203                    }
204            }
205    
206            /**
207             * Sets the base path this classloader will look for classes in.
208             * 
209             * @param basepath A folder path to look for classes.
210             */
211            public void setBasepath(final String basepath) {
212                    this.basepath = basepath;
213            }
214    
215            /**
216             * @return {@link #basepath}
217             */
218            public String getBasepath() {
219                    return basepath;
220            }
221    }
