001    package org.codemonkey.javareflection;
002    
003    import java.lang.reflect.Field;
004    import java.lang.reflect.Method;
005    import java.lang.reflect.Modifier;
006    import java.util.Arrays;
007    import java.util.EnumSet;
008    import java.util.HashMap;
009    import java.util.LinkedList;
010    import java.util.List;
011    import java.util.Map;
012    
013    import org.apache.commons.lang.StringUtils;
014    
015    /**
016     * A {@link Field} shorthand utility class mainly used to collect fields from classes meeting certain restrictions/requirements.
017     * <p>
018     * With this utility class you can perform field lookups, by combining lookup restriction criteria. A trivial example is find all fields on
019     * a class <em>Apple</em>, not looking at its super classes, which should be protected, have a getter method, but not a setter method:
020     * 
021     * <pre>
022     * FieldUtils.collectFields(Apple.class, Apple.class, EnumSet.of(Visibility.PROTECTED),
023     *              EnumSet.of(BeanRestriction.YES_GETTER, BeanRestriction.NO_SETTER));
024     * </pre>
025     * 
026     * @author Benny Bottema
027     * @see #collectFields(Class, Class, EnumSet, EnumSet)
028     */
029    public final class FieldUtils {
030            private FieldUtils() {
031            }
032    
033            /**
034             * Determines what visibility modifiers a field is allowed to have in {@link FieldUtils#collectFields(Class, Class, EnumSet, EnumSet)}.
035             * 
036             * @author Benny Bottema
037             */
038            public enum Visibility {
039                    /**
040                     * Visibility flag that corresponds with java's keyword <code>private</code>.
041                     */
042                    PRIVATE(Modifier.PRIVATE),
043                    /**
044                     * Visibility flag that corresponds with java's visibility modifier <code>default</code> (package protected).
045                     */
046                    DEFAULT(-1), // no Java equivalent
047                    /**
048                     * Visibility flag that corresponds with java's keyword <code>protected</code>.
049                     */
050                    PROTECTED(Modifier.PROTECTED),
051                    /**
052                     * Visibility flag that corresponds with java's keyword <code>public</code>.
053                     */
054                    PUBLIC(Modifier.PUBLIC);
055                    private int modifierFlag;
056    
057                    private Visibility(final int modifierFlag) {
058                            this.modifierFlag = modifierFlag;
059                    }
060            }
061    
062            /**
063             * Indicates whether a field needs a Bean setter or getter, exactly none or any combination thereof. Determines what kind of fields are
064             * potential collection candidates.
065             * 
066             * @author Benny Bottema
067             */
068            public enum BeanRestriction {
069                    /**
070                     * Restriction flag that indicates a <em>getter</em> method is required.
071                     */
072                    YES_GETTER,
073                    /**
074                     * Restriction flag that indicates a <em>setter</em> method is required.
075                     */
076                    YES_SETTER,
077                    /**
078                     * Restriction flag that indicates no <em>setter</em> must be available.
079                     */
080                    NO_SETTER,
081                    /**
082                     * Restriction flag that indicates a <em>getter</em> must be available.
083                     */
084                    NO_GETTER;
085            }
086    
087            /**
088             * Returns a pool of {@link Field} wrappers including optional relevant setter/getter methods, collected from the given class tested
089             * against the given visibility and Bean restriction requirements.
090             * <p>
091             * The returned fields are mapped against the classes they were found on, since field names can be declared multiple times with the same
092             * name.
093             * 
094             * @param _class The class (and chain) to harvest fields from.
095             * @param boundaryMarker The last <code>class></code> or <code>interface</code> implementing class that fields are collected from. Can
096             *            be used to prevent finding fields on a supper class.
097             * @param visibility A set of visibility requirements (ie. {@link Visibility#PROTECTED} indicates a field is allowed to have
098             *            <code>protected</code> visibility).
099             * @param beanRestrictions A set of Bean restriction requirements indicating a field should or shouldn't have a setter, getter or both.
100             * @return A Map per class in the chain with the fields declared by that class.
101             * @see #meetsVisibilityRequirements(Field, EnumSet)
102             * @see #resolveBeanProperty(Field, EnumSet)
103             */
104            public static Map<Class<?>, List<FieldWrapper>> collectFields(final Class<?> _class, final Class<?> boundaryMarker,
105                            final EnumSet<Visibility> visibility, final EnumSet<BeanRestriction> beanRestrictions) {
106                    final Map<Class<?>, List<FieldWrapper>> fields = new HashMap<Class<?>, List<FieldWrapper>>();
107                    final Field[] allFields = _class.getDeclaredFields();
108                    final List<FieldWrapper> filteredFields = new LinkedList<FieldWrapper>();
109                    for (final Field field : allFields) {
110                            if (meetsVisibilityRequirements(field, visibility)) {
111                                    final FieldWrapper property = resolveBeanProperty(field, beanRestrictions);
112                                    if (property != null) {
113                                            filteredFields.add(property);
114                                    }
115                            }
116                            fields.put(_class, filteredFields);
117                    }
118                    // determine if we need to look deeper
119                    final List<Class<?>> interfaces = Arrays.asList(_class.getInterfaces());
120                    if (_class.equals(boundaryMarker) || interfaces.contains(boundaryMarker)) {
121                            return fields;
122                    } else {
123                            fields.putAll(collectFields(_class.getSuperclass(), boundaryMarker, visibility, beanRestrictions));
124                            return fields;
125                    }
126            }
127    
128            /**
129             * Determines if the visibility modifiers of a given {@link Field} is included in the set of flags.
130             * 
131             * @param field The field who's visibility modifiers we want to test.
132             * @param visibility List of {@link Visibility} flags to test against.
133             * @return Whether a given field has one of the specified visibility flags.
134             */
135            static boolean meetsVisibilityRequirements(final Field field, final EnumSet<Visibility> visibility) {
136                    for (final Visibility visibilityModifier : visibility) {
137                            final int m = field.getModifiers();
138                            if (!visibilityModifier.equals(Visibility.DEFAULT)) {
139                                    if ((m & visibilityModifier.modifierFlag) != 0) {
140                                            return true;
141                                    }
142                            } else {
143                                    if (!Modifier.isPrivate(m) && !Modifier.isProtected(m) && !Modifier.isPublic(m)) {
144                                            return true;
145                                    }
146                            }
147                    }
148                    return false;
149            }
150    
151            /**
152             * Determines if a given <code>Field</code> meets the specified Bean restriction requirements and returns the field as a BeanProperty
153             * with optional getter/setter.
154             * 
155             * @param field The field to test.
156             * @param beanRestrictions The Bean restrictions to apply (should/shouldn't have setter/getter).
157             * @return Whether the field fits the restrictions.
158             */
159            static FieldWrapper resolveBeanProperty(final Field field, final EnumSet<BeanRestriction> beanRestrictions) {
160                    if (beanRestrictions.containsAll(EnumSet.of(BeanRestriction.NO_GETTER, BeanRestriction.YES_GETTER)) //
161                                    || beanRestrictions.containsAll(EnumSet.of(BeanRestriction.NO_SETTER, BeanRestriction.YES_SETTER))) {
162                            throw new IllegalArgumentException("cannot both include and exclude a setter/getter requirement");
163                    }
164    
165                    // since PropertyUtilsBean#getPropertyDescriptors(...) doesn't detect setters without getters (Bean convention)
166                    // we'll just find setter/getters manually
167                    final String setterName = "set" + StringUtils.capitalize(field.getName());
168                    final String getterName;
169                    if (field.getType().equals(boolean.class)) {
170                            getterName = "is" + StringUtils.capitalize(field.getName());
171                    } else {
172                            getterName = "get" + StringUtils.capitalize(field.getName());
173                    }
174                    final Method writeMethod = JReflect.findSimpleCompatibleMethod(field.getDeclaringClass(), setterName, field.getType());
175                    final Method readMethod = JReflect.findSimpleCompatibleMethod(field.getDeclaringClass(), getterName);
176    
177                    if (!((readMethod != null && beanRestrictions.contains(BeanRestriction.NO_GETTER)) //
178                                    || (!(readMethod != null) && beanRestrictions.contains(BeanRestriction.YES_GETTER)) //
179                                    || (writeMethod != null && beanRestrictions.contains(BeanRestriction.NO_SETTER)) //
180                    || (!(writeMethod != null) && beanRestrictions.contains(BeanRestriction.YES_SETTER)))) {
181                            return new FieldWrapper(field, readMethod, writeMethod);
182                    } else {
183                            return null;
184                    }
185            }
186    }
