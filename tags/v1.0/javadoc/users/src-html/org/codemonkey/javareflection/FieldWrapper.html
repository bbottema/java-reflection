001    package org.codemonkey.javareflection;
002    
003    import java.lang.reflect.Field;
004    import java.lang.reflect.Method;
005    
006    /**
007     * A wrapper class that keeps the property and its setter/getter in one place.
008     * 
009     * @author Benny Bottema
010     */
011    public class FieldWrapper {
012    
013            private final Field field;
014            private final Method getter;
015            private final Method setter;
016    
017            /**
018             * Initializes the wrapper with field, getter and setter, all optional.
019             * 
020             * @param field
021             * @param getter
022             * @param setter
023             */
024            public FieldWrapper(final Field field, final Method getter, final Method setter) {
025                    this.field = field;
026                    this.getter = getter;
027                    this.setter = setter;
028            }
029    
030            /**
031             * @return {@link #field}.
032             */
033            public Field getField() {
034                    return field;
035            }
036    
037            /**
038             * @return {@link #getter}.
039             */
040            public Method getGetter() {
041                    return getter;
042            }
043    
044            /**
045             * @return {@link #setter}.
046             */
047            public Method getSetter() {
048                    return setter;
049            }
050    }
