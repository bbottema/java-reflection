013    /**
014     * This reflection utility class predicts (and converts) which types a specified value can be converted into. It can only do conversions of
015     * known 'common' types, which include:
016     * <ul>
017     * <li>Any {@link Number} type (Integer, Character, Double, byte, etc.)</li>
018     * <li><code>String</code></li>
019     * <li><code>Boolean</code></li>
020     * <li><code>Character</code></li>
021     * </ul>
022     * 
023     * @author Benny Bottema
024     * @see IncompatibleTypeException
025     */
026    public final class ValueConverter {
027    
028            /**
029             * List of common types that all other common types can always convert to. For example, <code>String</code> and <code>Integer</code> are
030             * basic common types and can be converted to any other common type.
031             */
032            static final List<Class<?>> COMMON_TYPES = Arrays.asList(new Class<?>[] { String.class, Integer.class, int.class, Float.class,
033                            float.class, Double.class, double.class, Long.class, long.class, Byte.class, byte.class, Short.class, short.class,
034                            Boolean.class, boolean.class, Character.class, char.class });
035    
036            /**
037             * A list of all primitive number types.
038             */
039            static final List<Class<?>> PRIMITIVE_NUMBER_TYPES = Arrays.asList(new Class<?>[] { byte.class, short.class, int.class, long.class,
040                            float.class, double.class });
041    
042            /**
043             * @param c The class to inspect.
044             * @return whether given type is a known common type.
045             */
046            public static boolean isCommonType(final Class<?> c) {
047                    return COMMON_TYPES.contains(c);
048            }
049    
050            /**
051             * Private constructor prevents from instantiating this utility class.
052             */
053            private ValueConverter() {
054                    // utility class
055            }
056    
057            /**
058             * Determines to which types the specified value (its type) can be converted to. Most common types can be converted to most other common
059             * types and all types can be converted into a String using {@link Object#toString()}.
060             * 
061             * @param c The input type to find compatible conversion output types for
062             * @return The list with compatible conversion output types.
063             */
064            public static List<Class<?>> collectCompatibleTypes(final Class<?> c) {
065                    if (isCommonType(c)) {
066                            return Collections.unmodifiableList(COMMON_TYPES);
067                    } else {
068                            // not a common type, we only know we're able to convert to String
069                            return Arrays.asList(new Class<?>[] { String.class });
070                    }
071            }
072    
073            /**
074             * Converts a list of values to their converted form, as indicated by the specified targetTypes.
075             * 
076             * @param args The list with value to convert.
077             * @param targetTypes The output types the specified values should be converted into.
078             * @param useOriginalValueWhenIncompatible Indicates whether an exception should be thrown for inconvertible values or that the original
079             *            value should be used instead.
080             * @return Array containing converted values where convertible or the original value otherwise.
081             * @throws IncompatibleTypeException Thrown when unable to convert and not use the original value.
082             */
083            public static Object[] convert(final Object[] args, final Class<?>[] targetTypes, boolean useOriginalValueWhenIncompatible)
084                            throws IncompatibleTypeException {
085                    if (args.length != targetTypes.length) {
086                            throw new IllegalStateException("number of target types should match the number of arguments");
087                    }
088                    final Object[] convertedValues = new Object[args.length];
089                    for (int i = 0; i < targetTypes.length; i++) {
090                            try {
091                                    convertedValues[i] = convert(args[i], targetTypes[i]);
092                            } catch (IncompatibleTypeException e) {
093                                    if (useOriginalValueWhenIncompatible) {
094                                            // simply take over the original value and keep converting where possible
095                                            convertedValues[i] = args[i];
096                                    } else {
097                                            throw e;
098                                    }
099                            }
100                    }
101                    return convertedValues;
102            }

104            /**
105             * Converts a single value into a target output datatype. Only input/output pairs should be passed in here according to the possible
106             * conversions as determined by {@link #collectCompatibleTypes(Class)}.<br />
107             * <br />
108             * First checks if the input and output types aren't the same. Then the conversions are checked for and done in the following order:
109             * <ol>
110             * <li>conversion to <code>String</code></li>
111             * <li>conversion to any {@link Number}</li>
112             * <li>conversion to <code>Boolean</code></li>
113             * <li>conversion to <code>Character</code></li>
114             * </ol>
115             * 
116             * @param value The value to convert.
117             * @param targetType The target datatype the value should be converted into.
118             * @return The converted value according the specified target datatype.
119             * @throws IncompatibleTypeException
120             */
121            public static Object convert(final Object value, final Class<?> targetType)
122                            throws IncompatibleTypeException {
123                    if (value == null) {
124                            return null;
125                    }
126                    final Class<?> valueType = value.getClass();
127    
128                    // 1. check if conversion is required to begin with
129                    if (targetType.isAssignableFrom(valueType)) {
130                            return value;
131                            // 2. check if we can simply use Object.toString() implementation
132                    } else if (targetType.equals(String.class)) {
133                            return value.toString();
134                            // 3. check if we can reuse conversion from String
135                    } else if (valueType.equals(String.class)) {
136                            return convert((String) value, targetType);
137                            // 4. check if we can reuse conversion from a Number subtype
138                    } else if (Number.class.isAssignableFrom(valueType) || isPrimitiveNumber(valueType)) {
139                            return convert((Number) value, targetType);
140                            // 4. check if we can reuse conversion from boolean value
141                    } else if (valueType.equals(Boolean.class) || valueType.equals(boolean.class)) {
142                            return convert((Boolean) value, targetType);
143                            // 5. check if we can reuse conversion from character
144                    } else if (valueType.equals(Character.class) || valueType.equals(char.class)) {
145                            return convert((Character) value, targetType);
146                    } else {
147                            throw new IncompatibleTypeException(value, valueType.toString(), targetType.toString());
148                    }
149            }
150    
151            /**
152             * Attempts to convert a {@link Number} to the target datatype.
153             * <p>
154             * <strong>NOTE: </strong> precision may be lost when converting from a wide number to a narrower number (say float to integer). These
155             * conversions are done by simply calling {@link Number#intValue()} and {@link Number#floatValue()} etc.
156             * 
157             * @param value The number to convert.
158             * @param targetType The target datatype the number should be converted into.
159             * @return The converted number.
160             * @throws IncompatibleTypeException
161             */
162            public static Object convert(final Number value, final Class<?> targetType)
163                            throws IncompatibleTypeException {
164                    if (value == null) {
165                            return null;
166                    }
167                    if (targetType.equals(String.class)) {
168                            return value.toString();
169                    } else if (targetType.equals(Integer.class) || targetType.equals(int.class)) {
170                            return value.intValue();
171                    } else if (targetType.equals(Boolean.class) || targetType.equals(boolean.class)) {
172                            // any non-zero number converts to true
173                            return value.intValue() > 0;
174                    } else if (targetType.equals(Float.class) || targetType.equals(float.class)) {
175                            return value.floatValue();
176                    } else if (targetType.equals(Double.class) || targetType.equals(double.class)) {
177                            return value.doubleValue();
178                    } else if (targetType.equals(Long.class) || targetType.equals(long.class)) {
179                            return value.longValue();
180                    } else if (targetType.equals(Byte.class) || targetType.equals(byte.class)) {
181                            return value.byteValue();
182                    } else if (targetType.equals(Short.class) || targetType.equals(short.class)) {
183                            return value.shortValue();
184                    } else if (targetType.equals(Character.class) || targetType.equals(char.class)) {
185                            return Character.forDigit(value.intValue(), 10);
186                    } else {
187                            throw new IncompatibleTypeException(value, value.getClass().toString(), targetType.toString());
188                    }
189            }
190    
191            /**
192             * Attempts to convert a <code>Boolean</code> to the target datatype.
193             * 
194             * @param value The boolean to convert.
195             * @param targetType The target datatype the boolean should be converted into.
196             * @return The converted boolean.
197             * @throws IncompatibleTypeException
198             */
199            @SuppressWarnings("unchecked")
200            public static Object convert(final Boolean value, final Class<?> targetType)
201                            throws IncompatibleTypeException {
202                    if (value == null) {
203                            return null;
204                    }
205                    if (targetType.equals(Boolean.class) || targetType.equals(boolean.class)) {
206                            return value;
207                    } else if (targetType.equals(String.class)) {
208                            return value.toString();
209                    } else if (Number.class.isAssignableFrom(targetType) || isPrimitiveNumber(targetType)) {
210                            return convertNumber(value ? "1" : "0", (Class<Number>) targetType);
211                    } else if (targetType.equals(Character.class) || targetType.equals(char.class)) {
212                            return value ? '1' : '0';
213                    }
214                    // Boolean value incompatible with targetType
215                    throw new IncompatibleTypeException(value, Boolean.class.toString(), targetType.toString());
216            }
217    
218            /**
219             * Attempts to convert a <code>Character</code> to the target datatype.
220             * 
221             * @param value The character to convert.
222             * @param targetType The target datatype the character should be converted into.
223             * @return The converted character.
224             * @throws IncompatibleTypeException
225             */
226            @SuppressWarnings("unchecked")
227            public static Object convert(final Character value, final Class<?> targetType)
228                            throws IncompatibleTypeException {
229                    if (value == null) {
230                            return null;
231                    }
232                    if (targetType.equals(String.class)) {
233                            return value.toString();
234                    } else if (targetType.equals(char.class) || targetType.equals(Character.class)) {
235                            return value;
236                    } else if (Number.class.isAssignableFrom(targetType) || isPrimitiveNumber(targetType)) {
237                            // convert Character to Number
238                            return convertNumber(String.valueOf(value), (Class<Number>) targetType);
239                    } else if (targetType.equals(Boolean.class) || targetType.equals(boolean.class)) {
240                            // convert Character to Boolean
241                            if (value.equals('0')) {
242                                    return false;
243                            } else if (value.equals('1')) {
244                                    return true;
245                            } else {
246                                    // Character incompatible with type Boolean
247                            }
248                    }
249                    // Character incompatible with type targetType
250                    throw new IncompatibleTypeException(value, Character.class.toString(), targetType.toString());
251            }
252    
253            /**
254             * Attempts to convert a <code>String</code> to the target datatype.
255             * 
256             * @param value The string to convert.
257             * @param targetType The target datatype the string should be converted into.
258             * @return The converted string.
259             * @throws IncompatibleTypeException
260             */
261            @SuppressWarnings("unchecked")
262            public static Object convert(final String value, final Class<?> targetType)
263                            throws IncompatibleTypeException {
264                    if (value == null) {
265                            return null;
266                    } else if (targetType.equals(String.class)) {
267                            return value;
268                    } else if (targetType.isEnum()) {
269                            return convertEnum(value, (Class<? extends Enum<?>>) targetType);
270                    } else if (value.length() > 0) {
271                            if (value.length() == 1) {
272                                    // Convert String as Character
273                                    if (targetType.equals(Character.class) || targetType.equals(char.class)) {
274                                            return value.charAt(0);
275                                    } else {
276                                            return convert(value.charAt(0), targetType);
277                                    }
278                            } else if (Number.class.isAssignableFrom(targetType) || isPrimitiveNumber(targetType)) {
279                                    return convertNumber(value, (Class<? extends Number>) targetType);
280                            } else if (targetType.equals(Boolean.class) || targetType.equals(boolean.class)) {
281                                    // convert String to Boolean
282                                    if (value.equalsIgnoreCase("false") || value.equalsIgnoreCase("0")) {
283                                            return false;
284                                    } else if (value.equalsIgnoreCase("true") || value.equalsIgnoreCase("1")) {
285                                            return true;
286                                    } else {
287                                            // String incompatible with type Boolean
288                                    }
289                            }
290                    }
291                    // String value incompatible with targetType
292                    throw new IncompatibleTypeException(value, value.getClass().toString(), targetType.toString());
293            }

295            /**
296             * Attempts to convert a <code>String</code> to an Enum instance, by mapping to the enum's name using
297             * {@link Enum#valueOf(Class, String)}.
298             * 
299             * @param value The value, which should be the name of one instance of the given enum.
300             * @param targetType The enum type to which which we'll try to convert.
301             * @return An enum of the given type, or <code>null</code> otherwise.
302             */
303            public static Object convertEnum(final String value, final Class<? extends Enum<?>> targetType) {
304                    if (value == null) {
305                            return null;
306                    }
307                    // /CLOVER:OFF
308                    try {
309                            // /CLOVER:ON
310                            return targetType.getMethod("valueOf", String.class).invoke(null, value);
311                            // /CLOVER:OFF
312                    } catch (final IllegalArgumentException e) {
313                            throw new IncompatibleTypeException(value, Enum.class.toString(), targetType.toString(), e);
314                    } catch (final SecurityException e) {
315                            throw new IncompatibleTypeException(value, Enum.class.toString(), targetType.toString(), e);
316                    } catch (final IllegalAccessException e) {
317                            throw new IncompatibleTypeException(value, Enum.class.toString(), targetType.toString(), e);
318                    } catch (final InvocationTargetException e) {
319                            throw new IncompatibleTypeException(value, Enum.class.toString(), targetType.toString(), e);
320                    } catch (final NoSuchMethodException e) {
321                            throw new IncompatibleTypeException(value, Enum.class.toString(), targetType.toString(), e);
322                    }
323                    // /CLOVER:ON
324    
325            }

327            /**
328             * Attempts to convert a <code>String</code> to the specified <code>Number</code> type.
329             * 
330             * @param value The string value which should be a number.
331             * @param numberType The <code>Class</code> type that should be one of <code>Number</code>.
332             * @return A {@link Number} subtype value converted from <code>value</code> (or <code>null</code> if value is <code>null</code>).
333             */
334            public static Object convertNumber(final String value, final Class<? extends Number> numberType) {
335                    if (value == null) {
336                            return null;
337                    }
338                    Validate.isTrue(Number.class.isAssignableFrom(numberType) || isPrimitiveNumber(numberType));
339                    if (NumberUtils.isNumber(value)) {
340                            if (Integer.class.equals(numberType) || int.class.equals(numberType) || Number.class.equals(numberType)) {
341                                    return Integer.parseInt(value);
342                            } else if (Byte.class.equals(numberType) || byte.class.equals(numberType)) {
343                                    return Byte.parseByte(value);
344                            } else if (Short.class.equals(numberType) || short.class.equals(numberType)) {
345                                    return Short.parseShort(value);
346                            } else if (Long.class.equals(numberType) || long.class.equals(numberType)) {
347                                    return Long.parseLong(value);
348                            } else if (Float.class.equals(numberType) || float.class.equals(numberType)) {
349                                    return Float.parseFloat(value);
350                            } else if (Double.class.equals(numberType) || double.class.equals(numberType)) {
351                                    return Double.parseDouble(value);
352                            } else if (BigInteger.class.equals(numberType)) {
353                                    return BigInteger.valueOf(Long.parseLong(value));
354                            } else if (BigDecimal.class.equals(numberType)) {
355                                    return BigDecimal.valueOf(Long.parseLong(value));
356                            } else {
357                                    // specified type incompatible with Number
358                                    throw new IncompatibleTypeException(value, value.getClass().toString(), numberType.toString());
359                            }
360                    } else {
361                            // specified value incompatible with Number
362                            throw new IncompatibleTypeException(value, value.getClass().toString(), numberType.toString());
363                    }
364            }

366            /**
367             * Returns whether a {@link Class} is a primitive number.
368             * 
369             * @param targetType The class to check whether it's a number.
370             * @return Whether specified class is a primitive number.
371             */
372            public final static boolean isPrimitiveNumber(final Class<?> targetType) {
373                    return PRIMITIVE_NUMBER_TYPES.contains(targetType);
374            }

376            /**
377             * This exception can be thrown in any of the conversion methods of {@link ValueConverter}, to indicate a value could not be converted
378             * into the target datatype. It doesn't mean a failed attempt at a conversion, it means that there was no way to convert the input value
379             * to begin with.
380             * 
381             * @author Benny Bottema
382             */
383            public static final class IncompatibleTypeException extends RuntimeException {
384                    private static final long serialVersionUID = -9234872336546L;
385    
386                    private static final String pattern = "error: unable to convert value '%s': '%s' to '%s'";
387    
388                    /**
389                     * @see RuntimeException#RuntimeException(String, Throwable)
390                     */
391                    public IncompatibleTypeException(final String message, final Exception e) {
392                            super(message, e);
393                    }
394    
395                    /**
396                     * @see RuntimeException#RuntimeException(String)
397                     */
398                    public IncompatibleTypeException(final Object value, final String className, final String targetName) {
399                            super(String.format(pattern, value, className, targetName));
400                    }
401    
402                    /**
403                     * @see RuntimeException#RuntimeException(String, Throwable)
404                     */
405                    public IncompatibleTypeException(final Object value, final String className, final String targetName,
406                                    final Exception nestedException) {
407                            super(String.format(pattern, value, className, targetName), nestedException);
408                    }
409            }
